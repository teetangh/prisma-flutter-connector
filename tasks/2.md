 Step 3: Design the SDK API (Flutter Developer Experience)

 SDK Initialization:
 final sdk = PrismaFlutterSDK(
   graphqlEndpoint: 'https://api.example.com/graphql',
   authToken: 'optional-token',
 );

 Query API:
 // List products
 final products = await sdk.products.list(
   filter: ProductFilter(priceUnder: 100),
   orderBy: ProductOrderBy.name_ASC,
 );

 // Get single product
 final product = await sdk.products.findUnique(id: '123');

 Mutation API:
 // Create order
 final order = await sdk.orders.create(
   data: CreateOrderInput(
     userId: 'user-123',
     items: [OrderItemInput(productId: 'prod-1', quantity: 2)],
   ),
 );

 Error Handling:
 try {
   final product = await sdk.products.findUnique(id: 'invalid');
 } on NotFoundException catch (e) {
   print('Product not found: ${e.message}');
 } on NetworkException catch (e) {
   print('Network error: ${e.message}');
 }
